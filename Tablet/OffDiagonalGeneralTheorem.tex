\begin{theorem}
For every \(\delta>0\), there exists a constant \(L\) such that for all positive integers \(s\ge L\) and \(k\ge Ls\),
\[
  r(s,k)\ge \left(\frac{k}{s}\right)^{(1-\delta)s}.
\]
\end{theorem}
