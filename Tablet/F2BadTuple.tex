\begin{definition}
For natural numbers \(p\) and \(k\), a \(2k\)-tuple
\[
  (a_1,b_1,\ldots,a_k,b_k)\in(\mathbb F_2^p)^{2k}
\]
is called bad if
\[
  \langle a_j,b_i\rangle=1
\]
whenever \(1\le j\le i\le k\). Equivalently, writing the tuple as a function
\[
  ab:\{1,\ldots,k\}\to \mathbb F_2^p\times\mathbb F_2^p,
\]
the condition is
\[
  \langle (ab(j))_1,(ab(i))_2\rangle=1
\]
for all indices \(j\le i\).
\end{definition}
