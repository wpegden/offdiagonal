\begin{definition}
For a loop graph \(G\) on \(V\), the complement \(\overline G\) is the loop graph on the same vertex set with
\[
  xy \in E(\overline G) \quad\Longleftrightarrow\quad xy \notin E(G).
\]
This complements loops and non-loops as well as ordinary edges and non-edges.
\end{definition}
